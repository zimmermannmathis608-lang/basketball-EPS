    <script>

        const ROTATIONS = [

            { tour: 1, arbitres: "7 & 8", matchs: [{ a: 1, b: 2, t: "A" }, { a: 3, b: 4, t: "B" }, { a: 5, b: 6, t: "C" }] },

            { tour: 2, arbitres: "4 & 6", matchs: [{ a: 7, b: 8, t: "A" }, { a: 1, b: 3, t: "B" }, { a: 2, b: 5, t: "C" }] },

            { tour: 3, arbitres: "3 & 8", matchs: [{ a: 4, b: 6, t: "A" }, { a: 2, b: 7, t: "B" }, { a: 1, b: 5, t: "C" }] },

            { tour: 4, arbitres: "1 & 2", matchs: [{ a: 3, b: 8, t: "A" }, { a: 4, b: 5, t: "B" }, { a: 6, b: 7, t: "C" }] }

        ];


        let tournamentData = {};

        let timers = {};


        function formatTime(seconds) {

            const m = Math.floor(seconds / 60);

            const s = seconds % 60;

            return `${m}:${s.toString().padStart(2, '0')}`;

        }


        function switchTab(tab) {

            const mContent = document.getElementById('content-matchs');

            const cContent = document.getElementById('content-classement');

            const mTab = document.getElementById('tab-matchs');

            const cTab = document.getElementById('tab-classement');

            if (tab === 'matchs') {

                mContent.classList.replace('hidden', 'block');

                cContent.classList.replace('block', 'hidden');

                mTab.className = "flex-1 py-4 font-black uppercase text-xs tracking-widest border-b-4 border-orange-600 text-orange-600";

                cTab.className = "flex-1 py-4 font-black uppercase text-xs tracking-widest border-b-4 border-transparent text-slate-400";

            } else {

                mContent.classList.replace('block', 'hidden');

                cContent.classList.replace('hidden', 'block');

                cTab.className = "flex-1 py-4 font-black uppercase text-xs tracking-widest border-b-4 border-orange-600 text-orange-600";

                mTab.className = "flex-1 py-4 font-black uppercase text-xs tracking-widest border-b-4 border-transparent text-slate-400";

            }

        }


        function init() {

            const saved = localStorage.getItem('basket_eps_chrono_v2');

            if (saved) {

                tournamentData = JSON.parse(saved);

            } else {

                ROTATIONS.forEach((r, rIdx) => {

                    r.matchs.forEach((m, mIdx) => {

                        const id = `${rIdx}_${mIdx}`;

                        tournamentData[id] = {

                            scores: { a: 0, b: 0, details: { 

                                a: { raq: 0, hr: 0, lou: 0, fau: 0 }, 

                                b: { raq: 0, hr: 0, lou: 0, fau: 0 } 

                            }},

                            history: [],

                            status: 'idle', 

                            timeLeft: 300 

                        };

                    });

                });

                save();

            }

            render();

        }


        function save() {

            localStorage.setItem('basket_eps_chrono_v2', JSON.stringify(tournamentData));

        }


        function toggleTimer(id) {

            const m = tournamentData[id];

            if (m.status === 'idle') {

                m.status = 'running';

                startTimerLoop(id);

            } else if (m.status === 'running') {

                m.status = 'finished';

                clearInterval(timers[id]);

            }

            save();

            render();

        }


        function startTimerLoop(id) {

            if (timers[id]) clearInterval(timers[id]);

            timers[id] = setInterval(() => {

                const m = tournamentData[id];

                if (m.status !== 'running') {

                    clearInterval(timers[id]);

                    return;

                }

                m.timeLeft--;

                if (m.timeLeft <= 0) {

                    m.timeLeft = 0;

                    m.status = 'finished';

                    clearInterval(timers[id]);

                }

                save();

                updateTimerUI(id);

                if (m.status === 'finished') render();

            }, 1000);

        }


        function updateTimerUI(id) {

            const timerEl = document.getElementById(`timer-${id}`);

            if (timerEl) {

                timerEl.innerText = formatTime(tournamentData[id].timeLeft);

            }

            renderLiveTicker(); // Mise à jour du bandeau live aussi

        }


        function addPoint(id, team, type) {

            const m = tournamentData[id];

            if (m.status !== 'running') return;

            

            if (type === 'fau') {

                const opponent = (team === 'a' ? 'b' : 'a');

                m.scores.details[team].fau++;

                m.scores[opponent] += 1;

            } else {

                const pointsMap = { 'raq': 100, 'hr': 10, 'lou': 1 };

                m.scores.details[team][type]++;

                m.scores[team] += pointsMap[type];

            }

            

            m.history.push({ team, type });

            save();

            render();

        }


        function undoTeam(id, team) {

            const m = tournamentData[id];

            if (m.status !== 'running') return;

            const lastIdx = [...m.history].reverse().findIndex(action => action.team === team);

            if (lastIdx === -1) return;

            const actualIdx = m.history.length - 1 - lastIdx;

            const action = m.history.splice(actualIdx, 1)[0];

            

            if (action.type === 'fau') {

                const opponent = (team === 'a' ? 'b' : 'a');

                m.scores.details[team].fau--;

                m.scores[opponent] -= 1;

            } else {

                const pointsMap = { 'raq': 100, 'hr': 10, 'lou': 1 };

                m.scores.details[action.team][action.type]--;

                m.scores[action.team] -= pointsMap[action.type];

            }

            save();

            render();

        }


        function renderLiveTicker() {

            // On cherche la rotation en cours (celle qui a au moins un match 'running')

            // Sinon on prend la première qui n'est pas 'finished'

            let activeRotationIdx = ROTATIONS.findIndex((r, rIdx) => 

                r.matchs.some((m, mIdx) => tournamentData[`${rIdx}_${mIdx}`].status === 'running')

            );

            

            if (activeRotationIdx === -1) {

                activeRotationIdx = ROTATIONS.findIndex((r, rIdx) => 

                    r.matchs.some((m, mIdx) => tournamentData[`${rIdx}_${mIdx}`].status === 'idle')

                );

            }

            

            if (activeRotationIdx === -1) activeRotationIdx = ROTATIONS.length - 1;


            const ticker = document.getElementById('live-ticker');

            const rotation = ROTATIONS[activeRotationIdx];

            

            ticker.innerHTML = rotation.matchs.map((m, mIdx) => {

                const id = `${activeRotationIdx}_${mIdx}`;

                const data = tournamentData[id];

                const statusColor = data.status === 'running' ? 'text-green-400' : data.status === 'finished' ? 'text-red-500' : 'text-slate-500';

                

                return `

                    <div class="flex flex-col items-center">

                        <div class="text-[7px] font-black text-slate-500 uppercase tracking-widest mb-0.5">Terrain ${m.t}</div>

                        <div class="flex items-center gap-3">

                            <span class="text-[9px] font-bold text-slate-300">Éq ${m.a}</span>

                            <span class="text-sm font-black score-font text-orange-500">${data.scores.a}</span>

                            <span class="text-[8px] font-black text-slate-600">-</span>

                            <span class="text-sm font-black score-font text-orange-500">${data.scores.b}</span>

                            <span class="text-[9px] font-bold text-slate-300">Éq ${m.b}</span>

                        </div>

                        <div class="text-[7px] font-black ${statusColor} italic mt-0.5">${data.status === 'running' ? formatTime(data.timeLeft) : data.status.toUpperCase()}</div>

                    </div>

                `;

            }).join('<div class="w-px h-8 bg-slate-800"></div>');

        }

        function renderRanking() {

            const stats = {};

            let globalTotal = 0, globalRaq = 0, globalHr = 0, globalLou = 0, globalFau = 0;


            for (let i = 1; i <= 8; i++) {

                stats[i] = { total: 0, raq: 0, hr: 0, lou: 0, fau: 0, matchsJous: 0 };

            }


            Object.entries(tournamentData).forEach(([key, val]) => {

                const [rIdx, mIdx] = key.split('_').map(Number);

                const config = ROTATIONS[rIdx].matchs[mIdx];

                ['a', 'b'].forEach(side => {

                    const tid = config[side];

                    stats[tid].total += val.scores[side];

                    stats[tid].raq += val.scores.details[side].raq;

                    stats[tid].hr += val.scores.details[side].hr;

                    stats[tid].lou += val.scores.details[side].lou;

                    stats[tid].fau += val.scores.details[side].fau;

                    if (val.status === 'finished') stats[tid].matchsJous++;

                    

                    globalTotal += val.scores[side];

                    globalRaq += val.scores.details[side].raq;

                    globalHr += val.scores.details[side].hr;

                    globalLou += val.scores.details[side].lou;

                    globalFau += val.scores.details[side].fau;

                });

            });


            const sorted = Object.keys(stats).map(k => ({ id: k, ...stats[k] })).sort((a, b) => b.total - a.total);
